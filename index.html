          let even=0;
          let odd=0;
          let res=[];
          for(let i=0;i< N;i++){
            if(Arr[i] % 2===0){
              even=even+Arr[i];
            }
            else{
              odd=odd+Arr[i];
            }
          }
          res[0]=even;
          res[1]=odd;
          return res;
        };

Min and Max

    function arrayMin(arr) {
            let min=arr[0];
           for(let i=0;i< arr.length;i++){
             if(arr[i]< min){
               min=arr[i];
             }
           }  
           return min;
         }
         
    function arrayMax(arr) {
          let max=arr[0];
           for(let i=0;i< arr.length;i++){
             if(arr[i]>max){
               max=arr[i];
             }
           }  
           return max;
         }
